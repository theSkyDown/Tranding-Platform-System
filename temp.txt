用户表 user
	获取所有用户功能（包含分页、关键字搜索功能）+
	用户添加功能 +
	用户信息管理功能(用户信息修改功能) +
	用户密码重置功能 +
	用户删除功能 +
	用户批量删除功能 +
	用户登陆功能 +
	用户详细信息查看功能 +
	用户密码修改功能 +
	用户状态切换功能 +

	字段：
		user_id int  	        用户标识
		username varchar  	用户名
		phone varchar           手机号码
		password varchar	用户密码
		gender int		性别
		role_id int		角色标识
		avatar varchar		头像
		create_time datetime	创建时间
		update_time datetime	最后更新时间
		status int		用户状态
		
角色表 role
	获取所有角色信息（分页）+
	角色添加功能 +
	角色删除功能 +
	角色批量删除功能 +  
	角色修改功能 +

	字段：
		role_id int		角色标识
		role_name varchar	角色名称
		create_time 		角色创建时间

资源表 module
	获取所有资源信息（分页）+
	资源的添加 +
	资源的删除 +
	资源的批量删除 +
	资源修改功能 +
	
	字段：
		module_id int	 	资源标识
		module_name varchar	资源名称
		url varchar		对应的访问路径
		is_menu int		是否为菜单
	
权限表 permission
	角色功能限制功能 +
	权限分配功能

	字段：
		permission_id int 	权限标识
		role_id int		角色id
		module_id int		资源id
		
商品表 goods
	商品管理功能
	商品购买功能	
	
	字段：
		good_id int 		商品标识
		good_name varchar	商品名称
		price double		商品价格
		user_id int		商品的发布者标识
		description varchar	商品的描述
		good_img varchar	商品的图片
		category_id int		商品的类别标识
		create_time datetime	创建时间
